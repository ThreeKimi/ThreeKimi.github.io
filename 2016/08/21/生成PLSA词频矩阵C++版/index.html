生成PLSA词频矩阵C++版

发表于 2016-08-21 | 分类于 技术 |
  • 以下代码所做的工作是从英文的词频文件（上一篇博文）生成NLP中提取主题模型的PLSA算法的C++版本实现，C++版本实现平均的处理时间快了30倍以上。
    #include <iostream>
    #include <string>
    #include <vector>
    #include <fstream>
    #include <map>

    using namespace std;

    /*
     @in, src: 待分割的字符串
     @in, delim: 分隔符字符串
     @in_out, dest: 保存分割后的每个字符串
     */

    void split(const string& src, const string& delim, vector<string>& dest)
    {
        string str = src;
        string::size_type start = 0, index;
        string substr;

    index = str.find_first_of(delim, start);    //在str中查找(起始：start) delim的任意字符的第一次出现的位置
    while(index != string::npos)
    {
        substr = str.substr(start, index-start);
        dest.push_back(substr);
        start = str.find_first_not_of(delim, index);    //在str中查找(起始：index) 第一个不属于delim的字符出现的位置
        if(start == string::npos) return;

        index = str.find_first_of(delim, start);
        }
    }


    int main()
    {
        ifstream infile("/Users/If/Desktop/test1.txt", ios::in); //输入文件

    ofstream outfile1("/Users/If/Desktop/word1000.txt",ios::out); //输出文件1总词表
    ofstream outfile2("/Users/If/Desktop/document1000.txt",ios::out); //输出文件2单词矩阵

    string word;
    string delim(" ");
    string textline;

    int index=0;
    vector<string> name;
    map<string,int> Dic; //纪录总词表
    int c_num=0;
    int n=0;
    int count=0;
    vector<map<int,int>> store;

    if(infile.good())
    {
        while(!infile.fail()){
            count=0;
            map<int,int> dic;
            cout<<"Process"<<c_num++<<"line"<<endl; //输出已处理一行

            vector<string> results; //保存每一行中的各个string
            int num=0; //单词在总词表的序号
            getline(infile, textline);
            split(textline, delim, results); //每行单词保存在results vector中
            if (results.size() == 0) {
                break;
            }
            string str=results[0];
            name.push_back(str);
            for(size_t i=1;i<results.size();++i){
                string word=results[i];
                if (word.size()>18) { //单词长度大于18
                    continue;
                }

                if(Dic.find(word) == Dic.end()){ //更新总词表
                    num=index;
                    Dic[word]=index++;
                }else{
                    num=Dic[word];
                }

                if(dic.find(num) == dic.end()){ //更新每行词表
                    dic[num]=1;
                    ++count;
                }else{
                    ++dic[num];
                }

            }
            store.push_back(dic);
        }
    }
    n=(int)Dic.size();
    outfile2<<c_num-1<<' '<<n<<endl;
    for (size_t i=0; i<name.size(); ++i) {
        outfile2<<name[i]<<' ';
    }
    outfile2<<endl;

    for (int i=0; i<n; ++i) {
        outfile2<<i<<' ';
    }

    outfile2<<endl;

    for (size_t i=0; i<store.size(); ++i){
        map<int,int> temp=store[i];
        outfile2<<i<<' '<<store[i].size()<<' ';

        for(map<int,int>::iterator it=temp.begin();it != temp.end();++it){
            outfile2<<it->first<<' '<<it->second<<' ';
        }
        outfile2<<endl;
    }
    cout<<"Last step"; //即将进行最后一步输出总词表,胜利在望

    for(map<string,int>::iterator it=Dic.begin();it != Dic.end();++it){
        outfile1<<it->first<<":"<<it->second<<endl;
    }

    infile.close();
    outfile1.close();
    outfile2.close();

    return 0;
}
