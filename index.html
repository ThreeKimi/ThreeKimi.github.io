PAT甲级 1014 思路总结

原题链接

主要思路

  • 黄线以内有空位时，后续客户依次进入，直到黄线内区域已满（或已无后续客户），则进入下一步。
  • vector>> 用来存储银行窗口排队的情况，每次选择每个窗口的最前面的目前需要服务时间最短的客户，让其处理完毕（可能有多个），纪录该序号客户处理完毕的时间（f_service），并更新其它窗口的队头客户所需服务时间。
  • 输出时间时需要按照题目所需格式。
  • 注意17:00开始不再处理新业务，当已经在处理中的客户要处理完毕。
#include <iostream>
#include <vector>
#include <queue>
#include <climits>

#define WORK_TIME 9

using namespace std;

//检查目前的各窗口的黄线区域内是否有新的排队空间
bool check_in(vector<queue<pair<int,int>>>& window,int M){ 
    for (int i=0; i<window.size(); ++i) {

        if (window[i].size()<M) {
            return true;
        }
    }
    return false;
}

//黄线区域内是否每个窗口都无人排队
bool check_in_all(vector<queue<pair<int,int>>>& window,int M){
    for (int i=0; i<window.size(); ++i) {

        if (window[i].size()>0) {
            return false;
        }
    }
    return true;
}

 //序号为num的客户进入黄线排队
void yellow_in(vector<queue<pair<int,int>>>& window,vector<int>& vec,int num,int M){
    int sequence=INT_MAX;
    int count=0;
    for (int i=0; i<window.size(); ++i) {

        if (window[i].size()<sequence) {
            sequence=window[i].size();
            count=i;
        }
    }
    window[count].push(make_pair(num, vec[num]));
    //cout<<"window "<<sequence<<" "<<window[sequence].size();
}

//目前各窗口排队队头需要处理时间最小的客户让其处理完毕，并更新其它窗口的情况
int process_shortest_serviece(vector<queue<pair<int,int>>>& window,vector<int>& f_service,int &out,int max,int time){
    int min=INT_MAX;
    for (int i=0; i<window.size(); ++i) {
        if (window[i].size() == 0) {
            continue;
        }
        if (window[i].front().second<min) {
            min=window[i].front().second;
        }
    }

    time=time+min;

    for (int i=0; i<window.size(); ++i) {
        if (window[i].size() == 0) {
            continue;
        }
        if (window[i].front().second == min) {
            if (window[i].front().first>max) {
                out=1;
            }
            f_service[window[i].front().first]=time;
            window[i].pop();
            if (time>=540) {
                while (!window[i].empty()) {
                    window[i].pop();
                }
            }
        }else{
            window[i].front().second -= min;
        }
    }
    return min;
}

//按照题目要求输出时间
void adapter(int time){

    if (time == 0) {
        cout<<"Sorry";
    }else{
        if (time/60<2) {
            cout<<"0";
        }
        cout<<time/60+8<<":";

        if (time%60<10) {
            cout<<"0";
        }
        cout<<time%60;
    }
}

int main(){
    int N=0,M=0,K=0,Q=0;
    cin>>N>>M>>K>>Q;
    vector<int> p_time(K+1,0);
    for (int i=1; i <= K; ++i) {
        cin>>p_time[i];
    }
    vector<int> f_service(K+1,0);
    vector<int> query(Q+1,0);
    int max_query=0;
    for(int i=1;i <= Q;++i){
        cin>>query[i];
        if (query[i]>max_query) max_query=query[i];
    }

    vector<queue<pair<int,int>>> window(N,queue<pair<int,int>>()); //各窗口队列
    int time=0;//总时间
    int num=1;
    while (1) {
        while(check_in(window,M)){
            //cout<<window[0].size()<<" "<<window[1].size()<<endl;
            if (num <= K && time<WORK_TIME*60){
                yellow_in(window,p_time,num++,M);
            }else{
                break;
            }
        }
        int out=0;
        int increment=process_shortest_serviece(window,f_service,out,max_query,time);//可能处理多个窗口,处理完窗口就有空额,并纪录每个处理完客户的时间（分钟表示）

        if (check_in_all(window, M) || out) {
            break;
        }
        time+=increment;
    }
    for (int i=1; i <= Q; ++i) {
        adapter(f_service[query[i]]);
        if (i<Q) {
            cout<<endl;
        }
    }
    return 0;
}

PAT甲级 1016 思路总结

原题链接

主要思路

  • 输入可以按自己的要求改为更简单的形式，此题就把 on-line off-line改为bool类型存储
  • 每一条电话记录包括用户姓名，时间，状态，我按照时间来进行排序，并同步更新姓名和状态的vector，即最终下标对应的姓名，时间，状态在排序后仍然对应同一个用户。由此带来一个问题，有没有更好的方式实现我这一步所做的内容。
  • 对name按字母序排序后放在new_name vector中，并去掉相同的人名（unique）,然后对于每一个用户依次判断有效通话记录（通过一个栈，栈空时必须是开始通话状态才能插入，栈不空时当前记录必须是结束通话）

  • 对于没有产生有效账单的用户，无输出，此坑被我踩了。

    #include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <sstream>
#include <stack>
#include <cstdio>
using namespace std;

//将输入的时间字符串转变成分钟形式，从每月1号0点开始
int adapter_minute(string& str){
    return ((stoi(str.substr(3,2))-1)*1440+60*stoi(str.substr(6,2))+stoi(str.substr(9,2)));
}

//name，status按照record的排序方式同步排序
void my_sort(vector<string>& name,vector<int>& record,vector<bool>& status){
    for (int i=0; i<record.size(); ++i) {
        int min=i;
        for (int k=i+1; k<record.size(); ++k) {
            if (record[k]<record[min]) {
                min=k;
            }
        }
        string temp_name=name[i];
        int temp_record=record[i];
        bool temp_status=status[i];
        name[i]=name[min];
        name[min]=temp_name;
        record[i]=record[min];
        record[min]=temp_record;
        status[i]=status[min];
        status[min]=temp_status;

    }
}

//按题目要求输出时间
void adapter_time(int time){
    printf("%02d:",time/1440+1);
    time=time%1440;
    printf("%02d:%02d ",time/60,time%60);
}

//根据通话的起始时间计算费用
int money(int begin,int end,vector<int>& cost){
    int money=0;
    int sequence=(begin/60)%24;
    int remain=(begin%60)*cost[sequence];
    begin=(begin/60)*60;
    int hour=(end-begin)/60;
    int minute=(end-begin)%60;
    for (int i=0; i<hour; ++i) {
        money+=cost[sequence]*60;
        sequence=(sequence+1)%24;
    }
    money+=minute*cost[sequence];
    money-=remain;
    return money;
}

//查询并输出该查询人的账单
void print_bill(vector<int>& cost,string& query_name,vector<string>& name,vector<int>& record,vector<bool>& status,int month){
    stack<int> S;
    int total=0;
    int flag=0;
    for(int i=0;i<name.size();++i){
        if(name[i] == query_name){
            if(S.empty() && status[i] == true){
                S.push(record[i]);
            }else if(!S.empty() && status[i] == true){
                S.pop();
                S.push(record[i]);
            }else if(!S.empty() && status[i] == false){
                if (flag == 0){
                    flag=1;
                    printf("%s %02d\n",query_name.c_str(),month);
                }
                adapter_time(S.top());
                adapter_time(record[i]);
                cout<<record[i]-S.top()<<" $";
                int pay=money(S.top(),record[i],cost);
                total+=pay;
                printf("%.2f\n",pay/100.0);
                S.pop();
            }
        }
    }
    //账单不空
    if (flag) {
        cout<<"Total amount: $";
        printf("%.2f\n",total/100.0);
    }
}

int main(){
    vector<int> cost(24,0);
    for (int i=0; i<24; ++i) {
        cin>>cost[i];
    }

    int N=0;
    cin>>N;
    if(N == 0 || N == 1) return 0;
    vector<string> name(N);
    vector<int> record(N);
    vector<bool> status(N,0);
    int month=0;
    for (int i=0; i<N; ++i) {
        string str;
        cin>>name[i];
        cin>>str;
        month=stoi(str.substr(0,2));
        record[i]=adapter_minute(str);
        cin>>str;
        if (str == "on-line") status[i]=1;
    }

    vector<string> new_name=name;
    sort(new_name.begin(), new_name.end());

    //自定义的排序，name和status与时间作同等交换
    my_sort(name,record,status);

    //客户总数
    int name_num=(int)distance(new_name.begin(),unique(new_name.begin(), new_name.end()));
    for (int i=0; i<name_num; ++i) {
            print_bill(cost,new_name[i],name,record,status,month);
    }

    return 0;
}

亚马逊起步时仅仅只是一个网上书店，贝佐斯对这个网站的定位也是如此，这就要谈到第一个everything了，贝佐斯的策略是市面上任何一个出版社出的每一本书都要能从亚马逊买到，出版社每出一本新书，亚马逊就要存两本。亚马逊最终也没能做到这一点，但可以买到的书的数量已经远超其它竞争对手。
